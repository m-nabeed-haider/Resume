\newcommand{\ResumeHeader}[6]
{
    \begin{center}

        {\Huge\bfseries #1}

        \vspace{0.3em}

        {\large\textbf{#2}}

        \vspace{0.5em}

        #3

        \vspace{0.3em}

        #4 \;|\; #5 \;|\; #6

    \end{center}

    \vspace{1em}
}

% Project
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\ResumeProject}[3]
{
    \noindent
    \begin{tabularx}{\textwidth}{Xr}
        \textbf{#1} & {\small #3} \\
        \multicolumn{2}{l}{\textit{#2}}
    \end{tabularx}

    \vspace{4pt}
}
